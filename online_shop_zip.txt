1️⃣ index.html (asosiysi)
```
<!DOCTYPE html>
<html lang="uz">
<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<title>Online Do'kon</title>
<link rel="stylesheet" href="style.css">
</head>
<body>

<h2>Online Do'kon</h2>
<div class="category-select">
  <label for="category">Kategoriya:</label>
  <select id="category" onchange="filterProducts()">
    <option value="all">Barchasi</option>
    <option value="erkak">Erkaklar kiyimi</option>
    <option value="ayol">Ayollar kiyimi</option>
    <option value="bola">Bolalar kiyimi</option>
  </select>
</div>
<div class="products" id="products"></div>
<div class="cart" id="cart">
  <h3>Savat</h3>
  <ul id="cartItems"></ul>
  <p><strong>Jami: <span id="total">0</span> UZS</strong></p>
</div>
<script>
let productsData = [];
let cart = [];
let filteredProducts = [];
async function loadProducts(){
  const response = await fetch('products.json');
  productsData = await response.json();
  filteredProducts = [...productsData];
  filterProducts();
}
function displayProducts() {
  const container = document.getElementById("products");
  container.innerHTML = "";
  filteredProducts.forEach(p => {
    const div = document.createElement("div");
    div.className = "product";
    div.innerHTML = `
      <img src="${p.img}" alt="${p.name}">
      <h4>${p.name}</h4>
      <p>${p.price} UZS</p>
      <input type="number" id="qty${p.id}" value="1" min="1">
      <button onclick="addToCart(${p.id})">Savatga qo'shish</button>
    `;
    container.appendChild(div);
  });
}
function filterProducts(){
  const category = document.getElementById("category").value;
  filteredProducts = category === "all" ? [...productsData] : productsData.filter(p => p.category === category);
  displayProducts();
}
function addToCart(id){
  const qty = parseInt(document.getElementById(`qty${id}`).value);
  const product = productsData.find(p => p.id===id);
  const existing = cart.find(item => item.id===id);
  if(existing) existing.qty += qty;
  else cart.push({...product, qty});
  updateCart();
}
function removeFromCart(id){
  cart = cart.filter(item => item.id !== id);
  updateCart();
}
function updateCart(){
  const cartItems = document.getElementById("cartItems");
  cartItems.innerHTML = "";
  cart.forEach(item => {
    const li = document.createElement("li");
    li.innerHTML = `${item.name} x ${item.qty} - ${item.price*item.qty} UZS 
                    <button onclick="removeFromCart(${item.id})">❌</button>`;
    cartItems.appendChild(li);
  });
  const total = cart.reduce((sum, item) => sum + item.price*item.qty, 0);
  document.getElementById("total").textContent = total;
}
loadProducts();
</script>
</body>
</html>
```
